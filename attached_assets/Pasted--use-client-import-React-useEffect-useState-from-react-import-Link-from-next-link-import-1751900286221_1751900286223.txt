'use client'

import React, { useEffect, useState } from 'react'
import Link from 'next/link'
import { MobileMenu } from './MobileMenu'
import Button from '@/components/ui/button'

export default function Navigation() {
  const [scrolled, setScrolled] = useState(false)
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false)

  useEffect(() => {
    const handleScroll = () => {
      setScrolled(window.scrollY > 100)
    }
    window.addEventListener('scroll', handleScroll)
    return () => window.removeEventListener('scroll', handleScroll)
  }, [])

  return (
    <>
      <div className={`nav-bar ${scrolled ? 'scrolled' : ''}`}>
        <nav className="nav-container">
          <div className="nav-logo">SSELFIE STUDIO</div>
          
          {/* Desktop Menu - Revolutionary Platform Navigation */}
          <ul className="nav-menu">
            <li><Link href="/how-it-works" style={{color: '#FFFFFF'}}>How It Works</Link></li>
            <li><Link href="/pricing" style={{color: '#FFFFFF'}}>Pricing</Link></li>
            <li><Link href="/about" style={{color: '#FFFFFF'}}>About</Link></li>
            <li><Link href="/login" style={{color: '#FFFFFF'}}>Login</Link></li>
          </ul>

          {/* Mobile Menu Toggle */}
          <Button
            variant="ghost"
            className="md:hidden"
            onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
            aria-label="Toggle mobile menu"
          >
            â˜°
          </Button>
        </nav>
      </div>

      {/* Mobile Menu Component - Only show on mobile when open */}
      {mobileMenuOpen && (
        <MobileMenu
          isOpen={mobileMenuOpen}
          onClose={() => setMobileMenuOpen(false)}
          isAuthenticated={false}
          user={undefined}
        />
      )}

      <style jsx>{`
        /* Navigation */
        .nav-bar {
          position: fixed;
          top: 0;
          left: 0;
          right: 0;
          z-index: 1000;
          background: transparent;
          transition: all 0.3s ease;
        }
        
        .nav-bar.scrolled {
          background: rgba(255, 255, 255, 0.1);
          backdrop-filter: blur(10px);
        }
        
        .nav-bar.scrolled .nav-logo,
        .nav-bar.scrolled .nav-menu a {
          color: #FFFFFF !important;
        }
        
        .nav-container {
          padding: 32px 60px;
          display: flex;
          justify-content: space-between;
          align-items: center;
          color: #FFFFFF;
        }
        
        .nav-logo {
          font-family: 'Inter', sans-serif;
          font-size: 12px;
          font-weight: 300;
          letter-spacing: 0.3em;
          text-transform: uppercase;
          color: #FFFFFF;
          transition: color 0.3s ease;
        }
        
        .nav-menu {
          display: flex;
          gap: 48px;
          list-style: none;
          color: #FFFFFF;
        }
        
        .nav-bar .nav-menu a {
          font-size: 11px;
          letter-spacing: 0.2em;
          text-transform: uppercase;
          text-decoration: none;
          color: #FFFFFF !important;
          opacity: 0.8;
          transition: all 0.3s ease;
        }
        
        .nav-bar .nav-menu a:hover {
          opacity: 1;
        }
        
        /* Mobile Menu Toggle */
        .mobile-menu-toggle {
          display: none;
          flex-direction: column;
          gap: 4px;
          background: none;
          border: none;
          cursor: pointer;
          padding: 8px;
        }
        
        .hamburger-line {
          width: 24px;
          height: 2px;
          background-color: #FFFFFF;
          transition: all 0.3s ease;
        }
        
        /* Force hide mobile menu on desktop */
        @media (min-width: 1024px) {
          #mobile-menu {
            display: none !important;
          }
        }
        
        /* Mobile Responsive */
        @media (max-width: 1024px) {
          .nav-container {
            padding: 24px 40px;
          }
        }
        
        @media (max-width: 768px) {
          .nav-container {
            padding: 20px;
          }
          
          .nav-menu {
            display: none;
          }
          
          .mobile-menu-toggle {
            display: flex;
          }
        }
      `}</style>
    </>
  )
}
