    <script>
        function startPhotoshoot() {
            // Smooth scroll to main chat area
            const heroHeight = document.querySelector('.hero').offsetHeight;
            window.scrollTo({
                top: heroHeight,
                behavior: 'smooth'
            });
        }

        let isConversationStarted = false;

        function newSession() {
            // Reset to welcome state
            isConversationStarted = false;
            document.getElementById('welcome-state').style.display = 'block';
            document.getElementById('conversation').style.display = 'none';
            document.getElementById('message-input').value = '';
        }

        function loadSession(sessionId) {
            // Hide welcome state and show conversation
            document.getElementById('welcome-state').style.display = 'none';
            document.getElementById('conversation').style.display = 'block';
            isConversationStarted = true;
            
            // Highlight selected session
            document.querySelectorAll('.session-item').forEach(item => {
                item.style.background = 'transparent';
            });
            event.target.closest('.session-item').style.background = 'rgba(10, 10, 10, 0.1)';
        }

        function selectQuickStyle(style) {
            // Start conversation with selected style
            document.getElementById('welcome-state').style.display = 'none';
            document.getElementById('conversation').style.display = 'block';
            isConversationStarted = true;
            
            // Add visual feedback
            event.target.style.transform = 'scale(1.1)';
            event.target.style.borderColor = 'var(--black)';
            
            // Simulate sending the style choice
            setTimeout(() => {
                const styleNames = {
                    'editorial': 'Editorial luxury - magazine sophistication',
                    'natural': 'Natural beauty - effortless elegance',
                    'professional': 'Professional power - CEO energy',
                    'creative': 'Creative artistic - unique expression',
                    'lifestyle': 'Lifestyle natural - authentic moments',
                    'confident': 'Confident power - unapologetic strength'
                };
                
                // Focus the input and add the style choice
                document.getElementById('message-input').value = styleNames[style] || 'I want to explore this style';
            }, 300);
        }

        function sendMessage() {
            const input = document.getElementById('message-input');
            const message = input.value.trim();
            
            if (!message) return;
            
            if (!isConversationStarted) {
                // Start conversation
                document.getElementById('welcome-state').style.display = 'none';
                document.getElementById('conversation').style.display = 'block';
                isConversationStarted = true;
            }
            
            // Clear input
            input.value = '';
            
            // Show typing indicator
            document.getElementById('typing-indicator').style.display = 'flex';
            
            // Simulate Maya's response after delay
            setTimeout(() => {
                document.getElementById('typing-indicator').style.display = 'none';
                // Here you would add the actual message to the conversation
            }, 2000);
        }

        // Auto-resize textarea
        document.getElementById('message-input').addEventListener('input', function() {
            this.style.height = 'auto';
            this.style.height = Math.min(this.scrollHeight, 120) + 'px';
        });

        // Send message on Enter (but not Shift+Enter)
        document.getElementById('message-input').addEventListener('keydown', function(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });
